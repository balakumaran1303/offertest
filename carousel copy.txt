$(document).ready(function() {
    $(".slick-carousel").each(function(index) {
        var noOfCards = $(this).hasClass('banner-carousel');
        if (noOfCards) {
            $(this).find(".slick-slider").slick({
                dots: true,
                infinite: false,
                slidesToShow: 1,
                slidesToScroll: 1,
                autoplay: false,
                prevArrow: $(this).find(".prev-button"),
                nextArrow: $(this).find(".next-button"),
            });
        } else {
            $(this).find(".slick-slider").slick({
                dots: false,
                infinite: false,
                autoplaySpeed: 6000,
                slidesToShow: 4,
                slidesToScroll: 1,
                autoplay: false,
                prevArrow: $(this).find(".prev-button"),
                nextArrow: $(this).find(".next-button"),
                responsive: [{
                        breakpoint: 1024,
                        settings: {
                            slidesToShow: 3,
                            slidesToScroll: 1,
                        }
                    },
                    {
                        breakpoint: 576,
                        settings: {
                            slidesToShow: 1,
                            slidesToScroll: 1,
                        }
                    }
                ]
            });
        }
    });
});

function accordian(_this) {
    if (!$(_this).parent().hasClass('open')) {
        $('.links-sec').children('.footer-list-group').removeClass('open');
    }
    $(_this).parent().toggleClass('open');
}

// coustome
function accessible_bootstrap_accordionCoustom() {
    var keys = new keyCodes();

    // Tabs nav links
    $(document).on('keydown', '.links-sec [data-toggle="collapse"]', function(e) {
        var currentBtn = $(this),
            currentCardWrapper = currentBtn.parents('.footer-list-group');


        function next() {
            $(currentCardWrapper.next().find('[data-toggle="collapse"]').focus().attr('data-target'));
        }

        function prev() {
            $(currentCardWrapper.prev().find('[data-toggle="collapse"]').focus().attr('data-target'));
        }


        if (e.which === keys.down) {
            e.preventDefault(); // Disable arrow key scrolling
            // $(currentBtn.attr('data-target')).collapse('hide');

            if (currentCardWrapper.next('.footer-list-group').length) {
                next();
            } else {
                prev();
            }

        } else if (e.which === keys.up) {
            e.preventDefault(); // Disable arrow key scrolling
            // $(currentBtn.attr('data-target')).collapse('hide');

            if (currentCardWrapper.prev('.footer-list-group').length) {
                prev();
            } else {
                next();
            }
        }
    });
}

    accessible_bootstrap_accordionCoustom();
});


$(() => {
  $('.slick-carousel').each(function (index) {

    	var noOfCards = $(this).hasClass('banner-carousel');
        var autoplay = eval($(this).find('.rotating-carousel').attr('data-transition'));
        var autoplaySpeed = eval($(this).find('.rotating-carousel').attr('data-carousel-speed'));

		var milliseconds = autoplaySpeed !== undefined ? autoplaySpeed : 6000;
        var transition = autoplay !== undefined ? autoplay : false;

        if (noOfCards) {
            const bannerCarousel = $(this).find(".slick-slider");
            initializeSlick(bannerCarousel, $(this), 1, 1, [], true, transition, false, milliseconds);
        } else {
            const offerCarousel = $(this).find(".slick-slider");
            initializeSlick(offerCarousel, $(this), 1, 1, carouselResponsiveBrkPnts(3, 4), false, transition, false, milliseconds);
        }
  });
});
